Response에서 정보 추출해내기
from urllib import request
resp = request.urlopen("http://www.google.com")
resp.geturl() # 'http://www.google.com'
resp.reason # 'OK'
resp.getcode() # 200
resp.getheaders() # response message의 헤더

urllib의 parse 알아보기
    • parse.urlparse :  url을 schema, path, parameter 등으로 잘라줌  
from urllib import parse
parse.urlparse('https://www.google.com/search?q=%EB%B0%95%EB%B3%B4%EC%98%81')
>> ParseResult(scheme='https', netloc='www.google.com', path='/search', params='', query='q=%EB%B0%95%EB%B3%B4%EC%98%81', fragment='')
    • parese.urljoin : 정확히 모르겠음
parse.urljoin('https://www.google.com/search?q=%EB%B0%95%EB%B3%B4%EC%98%81', '/search/about')
>> 'https://www.google.com/search/about'
    • parese.urlencode : dict를 url에 들어갈 수 있는 query string으로 바꿔줌 
parse.urlencode({'q':'파이썬'})
>> 'q=%ED%8C%8C%EC%9D%B4%EC%8D%AC'
    • parse.quote_plus : str → url str(?)
parse.quote_plus('파이썬')
>> '%ED%8C%8C%EC%9D%B4%EC%8D%AC'
    • parse.unquote_plus : url str → str
parse.unquote_plus('%EB%B0%95%EB%B3%B4%EC%98%81')
