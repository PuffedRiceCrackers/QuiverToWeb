Relationship

• relationship 자체는 db.relationship을 통해 나타냄
• foreign key 는 그래도 적어줘야 함

1 : N

    ◦ ​db.relationship의 uselist = True
    ◦ 여기서는 Address에만 foreign key가 있는게 맞아서 거기에만 적은 것
    ◦ Person.addresses
        - Person이 소유한 모든 address의 collection이 됨
        - db.relationship 은 ‘Address’ object가 연결된 ‘Person’ obj를 reference할 수 있도록 하라는 뜻
# 1
class Person(db.Model):
id = db.Column(db.Integer, primary_key=True)
name = db.Column(db.String(50), nullable=False)
addresses = db.relationship('Address', uselist=True, backref='person', lazy=True)
# backref할 때 person이라는 property name을 쓰란 말인 듯
# 다른 이름일 수도 있음
# N
class Address(db.Model):
id = db.Column(db.Integer, primary_key=True)
street = db.Column(db.String(120), nullable=False)
person_id = db.Column(db.Integer, db.ForeignKey('person.id'), nullable=False)
>> 광은 = Person(id=1, name="광은")
>> 대방 = Address(id=1, street="여의대방로", person_id=1)
>> 치평 = Address(id=2, street="치평로", person_id=1)
>>> 광은.addresses # [대방obj, 치평obj]
>>> 대방.person # 광은 obj반환
>>> 치평.person # 광은 obj반환

1 : 1

​    ◦ db.relationship의 uselist = False
    ◦ 이를 어길 시 sqlite3.IntegrityError: UNIQUE constraint failed


N : N

    ◦ Helper Table 필요
        • 공식 문서에 Helper Table은 Class가 아니라 db.Table로 정의할 것을 권장함
    ◦ A-B가 N:N을 가질 때, A중 하나가 없어지면, Helper Table 에서도 자동으로 없어진다고 함
tags = db.Table(
db.Column('tag_id', db.INTEGER, db.ForeignKey('tag.id'), primary_key=True),
db.Column('page_id', db.INTEGER, db.ForeignKey('page.id'), primary_key=True)
)
class Page(db.Model):
id = db.Column(db.INTEGER, primary_key=True)
tags = db.relationship('Tag', secondary=tags, lazy='subquery', backref=db.backref('pages', lazy=True))
class Tag(db.Model):
id = db.Column(db.INTEGER, primary_key=True)
association_table = Table('association', Base.metadata,
Column('left_id', Integer, ForeignKey('left.id')),
Column('right_id', Integer, ForeignKey('right.id'))
)
class Parent(Base):
__tablename__ = 'left'
id = Column(Integer, primary_key=True)
children = relationship(
"Child",
secondary=association_table,
back_populates="parents")
class Child(Base):
__tablename__ = 'right'
id = Column(Integer, primary_key=True)
parents = relationship(
"Parent",
secondary=association_table,
back_populates="children")
