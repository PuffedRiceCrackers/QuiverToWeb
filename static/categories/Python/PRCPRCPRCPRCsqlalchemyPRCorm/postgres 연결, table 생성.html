# -*- coding: utf-8 -*-
from sqlalchemy import create_engine
from sqlalchemy import Column, String
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
import psycopg2 as pg
import psycopg2.extras
db_string = "postgresql://postgres:1234@127.0.0.1:5432/cosmetics"
db = create_engine(db_string) # engine 생성
base = declarative_base() "⭐️"
# session : 우리 대신 connection 을 관리해 줄 것임
Session = sessionmaker(db) # session - engine 연결
session = Session() # session 생성
# Python Obj를 생성. declarative_base를 상속
class Film(base):
__tablename__ = 'films'
title = Column(String, primary_key=True) # primary_key를 하면 기본적으로 autoincrement
director = Column(String)
year = Column(String)
# base를 상속하는 table schema들을 examine하고, create sql 실행
base.metadata.create_all(db) # Table 로 정의한 것 까지 create 해줌
# 몇개만 create
# base.metadata.tables["test"].create(bind = db)
# Create
doctor_strange = Film(title="Doctor Strange", director="Scott Derrickson", year="2016")
session.add(doctor_strange)
session.commit()
# Read
films = session.query(Film)
for film in films:
print(film.title)
# Update
doctor_strange.title = "Some2016Film"
session.commit()
# Delete
session.delete(doctor_strange)
session.commit()
