사용자로부터 데이터 받기

① GET method
    • url 에 정보가 드러남
        print(request.args.get("s"))  # s라는 이름의 input, default=None
<!--method : 이 방법으로 정보를 넘겨주겠다-->
<!--action : 이 url에서 넘겨준 정보를 처리하겠다-->
<form method="GET" action="/example">
<input type="text" name="search">검색
<input type="submit">
<input type="file">
</form>
@app.route('/example', methods=["GET","POST"])
def example():
print(request.form.get("search",None)) #넘겨주지 않으면 None
return ('got info')
