if
    if (exist?), if (true?)
    존재론 적인 관점에서 객체의 존재가 있냐 없냐로 따져주기  
if a is not None:
if not []:
if not '':
if not 0.0:
print('true')
for
    for (iterable)    
    for 에 else는 반복문이 전체가 완전히 실행될 때 마지막에 실행됨 (break 등으로 멈췄으면 실행안됨)
for i,j in {'a':1,'b':2}.items():
print(i,j)
else:
print('completed')
참고로 for loop 에서 참조는, copy 가 아니라 실제 address를 통해 참조함
a = ['가asdfa', '나asdf', 'adfa다']
# 같은 address를 보여줌
for i in range(len(a)):
print(id(a[i]))
for item in a:
print(id(item))

try, except, else, finally
try:
a = 1/0
except ZeroDivisionError as e: # 에러가 나면 실행 (여러개 쓸 수 있음)
print(e)
except NameError: # nesting도 가능
try:
( )
except:
print( )
except: # 맨 마지막 except 가 default 에러
print('error')
else: # 에러가 안나면 실행
print('No error')
finally: # 에러가 나든 안나든 실행
print('Finally')

assert
    condition이 true면 넘어가고, false면 error msg를 보내며 종료    
a = [1,2,3,4,5,6]
for item in a:
assert item < 4, '너무 값이 큽니다'
print(item) # 1,2,3까지 출력 후 AssertionError내며 종료
