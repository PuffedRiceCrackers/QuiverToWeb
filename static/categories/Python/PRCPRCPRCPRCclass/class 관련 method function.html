class 관련 method/function

isinstance(obj, class)
    obj 가 class의 instance인가 (부모 class도 true로 나옴)
issubclass(class, class)
    앞의 class가 뒤의 subclass인지
class organism:
@classmethod
def metabolize(cls, thing):
print(f"An organism metabolizes {thing}")
class animal(organism):
@classmethod
def metabolize(cls, thing):
print(f"An animal metabolizes {thing}")
cat = animal()
isinstance(cat, animal) # True
isinstance(cat, organism) # True
issubclass(animal, organism) # True
issubclass(organism, animal) # False
methodName.__self__
    해당 method를 지니는 instance object
methodName.__func__
    해당 method의 func object
cat = animal()
cat.eat.__self__ # __main__.animal
cat.eat.__func__ # <function __main__.animal.eat(cls, thing)>
className.__bases__
    해당 class의 직접적인 상속 class 반납
class A:
pass
class B(A):
pass
class C(B):
pass
C.__bases__ # (__main__.B,)
super().method
    모든! 부모의 method를 부른다는데 내가 실험한바로는 그렇지 않음… (1개만 불림) 👏🏻
class Dad:
def __init__(self, name):
print(f"{name}'s dad")
class Mom:
def __init__(self, name):
print(f"{name}'s mom'")
class Me(Dad, Mom):
def __init__(self, name):
super().__init__(name)
gwangeun = Me('gwangeun') # gwangeun's dad
