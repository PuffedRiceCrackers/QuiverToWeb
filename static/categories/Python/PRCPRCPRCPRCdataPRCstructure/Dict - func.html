    Dict - func

• 추출
    ◦ key만 접근           : default
    ◦ value만 접근         : .values()
    ◦ key, value 둘다 접근 : .items()
myDict = {'고군':'고양이', '광은':'사람'}
for i in myDict:
print(i)
>> '고양이'
>> '사람'
for i in myDict.values():
print(i)
>> '고군'
>> '광은'
for i in myDict.items():
print(i)
>> ('고군', '고양이')
>> ('광은', '사람')
• dict.fromkeys(seq, value=None)
    • seq 를 활용하여, 새로운 dict를 만들어줌
    • seq 의 중복을 제거함 (생각해보면 당연한게, key로 들어갈 건데 key가 중복될 순 없잖아)
    • 사실 dict는 sequence type이 아니지만, dict를 넣어도 key를 가지고 만들어 줌
a = {'고군':'고양이', '광은':'사람'}
dict.fromkeys(a, 'temp') >> {'고군': 'temp', '광은': 'temp'}
b = ['고군', '광은', '고군']
dict.fromkeys(b, 'temp') >> {'고군': 'temp', '광은': 'temp'}
