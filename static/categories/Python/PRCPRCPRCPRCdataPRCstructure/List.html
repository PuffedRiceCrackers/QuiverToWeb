List 생성/삽입/삭제/address copy

    • 생성
myList = [] # empty list 만들기
myList = list(range(5)) # [0,1,2,3,4]
myList = list(range(7,10)) # [7,8,,...,9] # list를 붙여야함을 잊지 말자

        기본적으로 a[1]='apple' 이런식으로 추가할 수 없음을 기억할 것
a.append('apple') # 뒤에 추가
a.insert(1,'banana') # 특정위치에 추가 (밀리면서 추가됨)

    • 다른 iterable을 통해서 연장하고자 할 때 .extend
        ※ 참고로 iterable 만 넣어줘야 함 ( a.extend(3) 은 error뜸 )
        ※ 만약 a.extend(‘abc’) 라고 주어지면 [‘a’,’b’,’c’] 가 넣어져서 에러도 못 찾으므로 주의할 것)
a = [1,2,3]
b = [4,5,6]
a.extend(b)
a >> [1,2,3,4,5,6] # append 였으면 차원이 생겼을 건데 extend를 썼기 때문에
    • 삭제
        ◦ 특정 위치 삭제         : ​ del a[1] 
        ◦ 특정 위치를 삭제 후 반환 : ​ k = a.pop(1) 
        ◦ 마지막 위치를 삭제      : ​ k = a.pop()  
        ◦ 특정 원소를 삭제       : ​ a.remove('apple’) 
        ※ 중복해서 삭제해 주는 건 아니니까 주의한다
a = [1, 2, 3, 4]
del a[1], a[2] # del a[1] 한 후, 변화한 a 에 대해 del a[2] 를 시행함
b = [1, 1, 1, 3, 4]
b.remove(1) # 모든 b 를 지워주진 않음
    • 개수만 맞다면 요런식으로 var에 깔끔하게 assign하는 것도 가능
myList=[1,2,3]
a,b,c = myList # a=1, b=2, c=3
List +, *가 가능
a = [-1] * 3 # [-1, -1, -1]
a = [-1] + [1] # [-1, 1]
a = [1, 2] - [2] # ❌ -는 안됨
List Slicing
# a[x:y] : x는 포함되고 y는 포함 안됨
a = ['a','b','c','d']
print(a[:]) #['a', 'b', 'c', 'd']
print(a[0:]) #['a', 'b', 'c', 'd']
print(a[1:]) #['b', 'c', 'd']
print(a[:2]) #['a', 'b']
다행히도 없는 애를 슬라이스 하려고 해도 에러가 나진 않음
a = [0,1,2,3,4,5]
a[10:] >> []
a[10:0] >> [0]
a[4:8] >> [4,5]

② List Comprehension 을 써보자
percentile = [[0,0,0] for i in range(2)] # [[0, 0, 0], [0, 0, 0]]
연산
a = [1,2,3]

>> c = a + b # [1,2,3,4,5,6]
기타
① list의 개수 반환
numElement = len(myList)
② list가 어떤 element를 가지는 지 살펴보기

if 1 in a:

③ tuple <-> list 가능함
a = [1,2,3]
a = tuple(a) # (1,2,3)
a = list(a) # [1,2,3]
④ 특정 element의 list에서의 index 반환
myList = ['a','b','c']

⑤ list 의 type을 casting하기
# 그냥 int()하는건 안됨
a = ['1','2']
>> int(a) # error
# map함수 사용할 것
>> list(map(int, a))
⑥ list의 sorting
# list 변형
a = [3, 1, 4]
>> a.sort() # a = [1, 3, 4]
# sort 된 list를 따로 만들기
a = [3, 1, 4]
b = sorted(a)
>> a # a = [3, 1, 4]
>> b # b = [1, 3, 4]
⑦ list의 max, min 등도 가능
a = [1, 2, 3]
>> max(a) # 3
